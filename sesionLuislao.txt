-- Tabla usuario
CREATE TABLE usuario(
    id_usuario INT AUTO_INCREMENT PRIMARY KEY, 
    contrasena VARCHAR(32), 
    correo VARCHAR(100) NOT NULL UNIQUE
);

-- Tabla rol
CREATE TABLE rol(
    id_rol INT AUTO_INCREMENT PRIMARY KEY, 
    rol VARCHAR(30) NOT NULL
);

-- Tabla permiso
CREATE TABLE permiso(
    id_permiso INT AUTO_INCREMENT PRIMARY KEY, 
    permiso VARCHAR(30) NOT NULL
);

-- Verifica que rol y permiso ya est√©n creadas correctamente antes de seguir

-- Tabla rol_permiso
CREATE TABLE rol_permiso(
    id_rol INT NOT NULL, 
    id_permiso INT NOT NULL, 
    PRIMARY KEY (id_rol, id_permiso), 
    FOREIGN KEY (id_rol) REFERENCES rol(id_rol) ON DELETE CASCADE ON UPDATE CASCADE, 
    FOREIGN KEY (id_permiso) REFERENCES permiso(id_permiso) ON DELETE CASCADE ON UPDATE CASCADE
);

-- Tabla usuario_rol
CREATE TABLE usuario_rol(
    id_usuario INT NOT NULL, 
    id_rol INT NOT NULL, 
    PRIMARY KEY (id_usuario, id_rol), 
    FOREIGN KEY (id_usuario) REFERENCES usuario(id_usuario) ON DELETE CASCADE ON UPDATE CASCADE, 
    FOREIGN KEY (id_rol) REFERENCES rol(id_rol) ON DELETE CASCADE ON UPDATE CASCADE
);
